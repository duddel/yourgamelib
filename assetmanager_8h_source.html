51  template <class T>
52  void put(std::string name, T *obj)
53  {
54  // insert delete function for objects of type T, if not present
55  auto itDelFunc = m_delFuncs.find(typeid(T).hash_code());
56  if (itDelFunc == m_delFuncs.end())
57  {
58  m_delFuncs[typeid(T).hash_code()] =
59  [](std::uintptr_t ptr) { delete reinterpret_cast<const T *>(ptr); };
60  }
61 
62  // insert (or replace) obj into map for objects of type T
63  auto itT = m_ptrs[typeid(T).hash_code()].find(name);
64  if (itT != m_ptrs[typeid(T).hash_code()].end())
65  {
66  m_delFuncs[typeid(T).hash_code()](itT->second);
67  }
68  m_ptrs[typeid(T).hash_code()][name] = reinterpret_cast<std::uintptr_t>(obj);
69  }
