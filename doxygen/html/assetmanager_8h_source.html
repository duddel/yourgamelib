61 template <class T>
62 void insert(std::string name, T *obj)
63 {
64 if (obj == nullptr)
65 return;
66
67 // insert delete function for objects of type T, if not present
68 auto itDelFunc = m_delFuncs.find(typeid(T).hash_code());
69 if (itDelFunc == m_delFuncs.end())
70 {
71 m_delFuncs[typeid(T).hash_code()] =
72 [](std::uintptr_t ptr)
73 { delete reinterpret_cast<const T *>(ptr); };
74 }
75
76 // insert (or replace) obj into map for objects of type T
77 auto itT = m_ptrMaps[typeid(T).hash_code()].find(name);
78 if (itT != m_ptrMaps[typeid(T).hash_code()].end())
79 {
80 m_delFuncs[typeid(T).hash_code()](itT->second);
81 }
82 m_ptrMaps[typeid(T).hash_code()][name] = reinterpret_cast<std::uintptr_t>(obj);
83 }

178 void clear()
179 {
180 // for all maps holding different pointer types...
181 for (const auto &map : m_ptrMaps)
182 {
183 // ...delete all objects pointed to...
184 for (const auto &ptr : map.second)
185 {
186 // ...using the registered delete function
187 m_delFuncs[map.first](ptr.second);
188 }
189 }
190 m_ptrMaps.clear();
191 }
